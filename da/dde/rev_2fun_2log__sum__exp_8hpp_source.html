76 inline double log_sum_exp_as_double(const std::vector<var>& x) {
77  using std::exp;
78  using std::log;
79  double max = NEGATIVE_INFTY;
80  for (size_t i = 0; i < x.size(); ++i) {
81  if (x[i] > max) {
82  max = x[i].val();
83  }
84  }
85  double sum = 0.0;
86  for (size_t i = 0; i < x.size(); ++i) {
87  if (x[i] != NEGATIVE_INFTY) {
88  sum += exp(x[i].val() - max);
89  }
90  }
91  return max + log(sum);
92 }
