133  inline kernel_parts get_kernel_parts(
134  std::set<const operation_cl_base*>& generated, name_generator& name_gen,
135  const std::string& i, const std::string& j) const {
136  kernel_parts res{};
137  if (generated.count(this) == 0) {
138  this->var_name = name_gen.generate();
139  generated.insert(this);
140  std::string i_arg = i;
141  std::string j_arg = j;
142  derived().modify_argument_indices(i_arg, j_arg);
143  std::array<kernel_parts, N> args_parts = index_apply<N>([&](auto... Is) {
144  return std::array<kernel_parts, N>{
145  std::get<Is>(arguments_)
146  .get_kernel_parts(generated, name_gen, i_arg, j_arg)...};
147  });
148  res.body
149  = std::accumulate(args_parts.begin(), args_parts.end(), std::string(),
150  [](const std::string& a, const kernel_parts& b) {
151  return a + b.body;
152  });
153  res.args
154  = std::accumulate(args_parts.begin(), args_parts.end(), std::string(),
155  [](const std::string& a, const kernel_parts& b) {
156  return a + b.args;
157  });
158  kernel_parts my_part = index_apply<N>([&](auto... Is) {
159  return this->derived().generate(i, j,
160  std::get<Is>(arguments_).var_name...);
161  });
162  res.body += my_part.body;
163  res.args += my_part.args;
164  }
